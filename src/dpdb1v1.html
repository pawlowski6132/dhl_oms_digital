SQL Query (Replace SMRU with your site)

SELECT

      a.[Incident_No]

      ,a.[Sub_Incident_Number]

      ,b.[Recordable_Injury]

      ,a.[Person_Responsible_Master_Incident]

      ,a.[Date_and_Time_of_Occurrence]

      ,a.[Date_Reported_to_Supervisor]

      ,a.[DHL_Date_Created]

      ,a.[Created_By]

      ,a.[Location]

      ,a.[Incident_Description]

      ,a.[Description_Of_Incident_Location]

      ,a.[Sub_Incident_Type]

      ,a.[Date_of_Occurrence]

      ,a.[DWH_INSERT_DATE]

  FROM [logics_final].[logics_Incident_1396_Mgmt_Master_final]a

  Left Join

       [logics_final].[logics_Incident_3333_Personal_Injury_final]b

          ON a.[Sub_Incident_Number] = b.[Sub_Incident_Number]

  where a.location like '%SMRU0840%'
